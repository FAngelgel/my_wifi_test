#include "camera_interface.h"
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/gpio.h"
#include "lv_port_disp.h"
#include "lv_port_indev.h"


lv_obj_t *main_screen = NULL;
lv_group_t *main_group = NULL;
lv_obj_t * state_label;
lv_obj_t *camera_screen;  // Your Camera Mode
bool camera_active = false; // Flag to control the camera task

static void my_first_button_event_handler(lv_event_t *e)
{
    if (lv_event_get_code(e) == LV_EVENT_CLICKED)
    {
        static bool toggle = false;
        toggle = !toggle;

        // 1. Hardware Control
        gpio_set_level(GPIO_NUM_4, toggle ? 1 : 0);

        // 2. Visual Control (Show on screen!)
        if (toggle) {
            lv_label_set_text(state_label, "LED: ON");
            lv_obj_set_style_bg_color(lv_event_get_target(e), lv_color_hex(0xd81b60), 0);
        } else {
            lv_label_set_text(state_label, "LED: OFF");
            lv_obj_set_style_bg_color(lv_event_get_target(e), lv_color_hex(0xd500f9), 0);
        }
    }
}

void my_first_button(void)
{
    lv_obj_t * btn = lv_btn_create(main_screen);
    lv_obj_set_size(btn, 150, 60);
    lv_obj_align(btn, LV_ALIGN_CENTER, 0, -50);
    

    /* This is the label we will update */
    state_label = lv_label_create(btn);
    lv_label_set_text(state_label, "LED: OFF");
    lv_obj_center(state_label);
    lv_obj_add_event_cb(btn, my_first_button_event_handler, LV_EVENT_CLICKED, NULL);
}

static void lvgl_act(void *pvParameter)
{
    init_my_pins(); 
    lv_init();
    lv_port_disp_init();
    lv_port_indev_init();
    
    extern lv_indev_t *indev_keypad;
    main_group = lv_group_create();
    lv_group_set_default(main_group);

    if (indev_keypad) {
        lv_indev_set_group(indev_keypad, main_group);
    }

    main_screen = lv_obj_create(NULL);
    lv_obj_set_style_bg_color(main_screen, lv_color_hex(0xffffff), 0);
    my_first_button(); 

    lv_scr_load(main_screen);

    while (1) {
        lv_timer_handler();
        vTaskDelay(pdMS_TO_TICKS(10));
    }
}

void start_lvgl_app(void)
{
    xTaskCreatePinnedToCore(lvgl_act, "gui_app", 4096 * 8, NULL, 5, NULL, 1);
}

void init_my_pins(void)
{
    gpio_config_t io_conf = {
        .pin_bit_mask = (1ULL << GPIO_NUM_4),
        .mode = GPIO_MODE_OUTPUT,
        .pull_up_en = GPIO_PULLUP_DISABLE,
        .pull_down_en = GPIO_PULLDOWN_ENABLE,
        .intr_type = GPIO_INTR_DISABLE,
    };
    gpio_config(&io_conf);
}

// Create a descriptor for a 320x240 RGB565 image
static lv_img_dsc_t cam_dsc = {
    .header.always_zero = 0,
    .header.w = 320,
    .header.h = 240,
    .header.cf = LV_IMG_CF_TRUE_COLOR, // This tells LVGL to expect RGB565
    .data_size = 320 * 240 * 2,
    .data = NULL, // We will fill this with the camera pointer later
};

lv_obj_t * cam_img_obj; // The widget on the screen

void create_camera_widget(void) {
    // Create the image widget on the main screen
    cam_img_obj = lv_img_create(main_screen);
    lv_obj_set_size(cam_img_obj, 320, 240);
    lv_obj_align(cam_img_obj, LV_ALIGN_CENTER, 0, 50);

    state_label = lv_label_create(cam_img_obj);
    lv_label_set_text(state_label, "camera");
    lv_obj_center(state_label);
    lv_obj_add_event_cb(cam_img_obj, my_first_button_event_handler, LV_EVENT_CLICKED, NULL);
    
    // Set a placeholder (optional)
    lv_img_set_src(cam_img_obj, &cam_dsc); 
}

void camera_task(void *pvParameters) {
    while(1) {
        if(camera_active) {
            camera_fb_t * fb = esp_camera_fb_get();
            if(fb) {
                cam_dsc.data = fb->buf;
                lv_img_set_src(cam_img_obj, &cam_dsc);
                lv_obj_invalidate(cam_img_obj);
                esp_camera_fb_return(fb);
            }
        }
        vTaskDelay(pdMS_TO_TICKS(30)); // Limit to ~30 FPS
    }
}
